//true if 
int getNumOfDivisors(int num)
{
	int count=2;
	double num_sqrt=sqrt((double)num);
	for(int i=2;i<=num_sqrt;i++){
		if(num%i==0){
			count+=2;
		}
	}
	if(num_sqrt*num_sqrt==num){
		count--;
	}
	//cout<<"Num: "<<num<<endl;
	//cout<<"Count: "<<count<<endl;
	return count;
}
