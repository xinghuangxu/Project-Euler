//true if 
int getMaxProduct(string numstr)
{
	int max=0;
	int prod=1;
	for(int i=4;i<numstr.size();i++){
		prod=1;
		for(int k=i-4;k<=i;k++){
			prod*=numstr[k]-48;
		}
		if(prod>max){
			max=prod;
		}
	}
	return max;
}
